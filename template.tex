\theoremstyle{definition}

\newcommand{\maketitlepage}{
  % Store old parindent value
  \setlength{\oldparindent}{\parindent}
  \setlength{\oldparskip}{\parskip}
  \pagenumbering{roman}      % Roman numbering (starting with i (one)) until first main chapter
  \input{template/frontmatter/titlepage}
  \setlength{\parindent}{\oldparindent}
  \setlength{\parskip}{\oldparskip}
}

\newcommand{\makelists}{
  % Store old paragraph separator
  \setlength{\oldparskip}{\parskip}
  \setlength{\parskip}{0pt}

  % TABLEOF CONTENTS
  \newpage
  \tableofcontents

  % List of figures (add to table of contents)
  \cleardoublepage
  \addcontentsline{toc}{chapter}{\listfigurename}
  \listoffigures

  % List of tables (add to table of contents)
  \cleardoublepage
  \addcontentsline{toc}{chapter}{\listtablename}
  \listoftables

  % COVER PAGE, TITLE PAGE AND IMPRINT PAGE
  % START OF MAIN DOCUMENT
  \cleardoublepage
  \setcounter{page}{1}
  \pagenumbering{arabic}      % Arabic numbering starting from 1 (one)
  \setlength{\parskip}{\oldparskip}
}

\newcommand{\makebackmatter}{
  % REFERENCES / BIBLIOGRAPHY
  \cleardoublepage
  \addcontentsline{toc}{chapter}{Bibliography}
  \bibliographystyle{acm}
  \bibliography{references}

  % APPENDICES
  \cleardoublepage
  \appendix
  \setcounter{page}{1}
  \pagenumbering{Roman}  % Capitalized roman numbering starting from I (one)

  \input{template/backmatter/appendix_1}
}
